DISTINCT  ve COUNT Kullanımı

* Bir değişkenin o veri grubunda aldığı benzersiz verileri görmemize yarar. Yani o değişken kaç farklı değer almış ve bunlar nelerdir?

SELECT DISTINCT rental_rate FROM film;  		--rental_rate değişkeninin kaç farklı değer aldığını gösterir

SELECT DISTINCT rental_rate, rating FROM film;  -- rental_rate VE rating değerleri farklı olan verileri getirir.

* Count

SELECT COUNT(*) FROM actor
WHERE first_name = 'Nick';        			 -- adı nick olan kaç kişi olduğunu gösterir.

SELECT COUNT(*) FROM actor
WHERE first_name LIKE 'A%'        			-- ismi A harfi ile başlayan kaç kişi olduğunu gösterir.

SELECT COUNT(*) FROM actor;      			 --actor tablosunda kaç veri olduğunu gösterir

SELECT COUNT( DISTINCT first_name ) FROM actor; -- ismi birbirinden farklı kaç veri olduğunu getirir.

SELECT COUNT(DISTINCT length ) FROM film;   	--film tablosunda uzunluğu birbirinden farklı kaç film olduğunu gösterir.
